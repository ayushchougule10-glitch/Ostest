#include <stdio.h> 
#include <stdlib.h> 
#include <unistd.h> 
#include <string.h> 
 
int main() 
{ 
    int fd[2]; 
    pid_t pid; 
    char write_msg[50], read_msg[50]; 
 
    
    if (pipe(fd) == -1) 
    { 
        perror("Pipe failed"); 
        exit(1); 
    } 
 
    pid = fork(); 
 
    if (pid < 0) 
    { 
        perror("Fork failed"); 
        exit(1); 
    } 
 
    if (pid > 0) 
    {   
      
        close(fd[0]); 
 
       printf("Enter a message for the child: "); 
       fgets(write_msg, sizeof(write_msg), stdin); 
 
 
        write(fd[1], write_msg, strlen(write_msg) + 1); 
        close(fd[1]); 
    } 
    else 
    {   
        close(fd[1]); 
read(fd[0], read_msg, sizeof(read_msg)); 
printf("Child received: %s\n", read_msg); 
close(fd[0]); 
} 
return 0; 
}
